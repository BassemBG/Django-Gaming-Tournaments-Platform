		document.addEventListener("DOMContentLoaded", () => {
			const searchBox = document.getElementById("search-box");
			const equipmentCards = document.querySelectorAll(".flip-card");
		
			searchBox.addEventListener("input", () => {
				const searchValue = searchBox.value.toLowerCase();
		
				equipmentCards.forEach(card => {
					const description = card.querySelector(".flip-card-back p:nth-of-type(2)")?.textContent.toLowerCase() || "";
		
					if (description.includes(searchValue)) {
						card.style.display = "block"; // Show the card if it matches the search
					} else {
						card.style.display = "none"; // Hide the card if it doesn't match
					}
				});
			});
		});
